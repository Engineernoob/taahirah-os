const labOrchestrator = createAgent({
  name: "Operator",
  goals: ["triage briefs", "dispatch labs", "publish reports"],
  tools: [queueClient("redis"), gitOps(), notionPublisher()],
  policy: guardrails.allowlist([
    "create_experiment",
    "submit_results",
    "request_review",
  ]),
});

labOrchestrator.on("experiment:ready", async (payload) => {
  await reviewAgent.validate(payload);
  await reportAgent.publish(payload);
});
